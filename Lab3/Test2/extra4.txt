struct Operands
{
	int o1;
	int o2;
};


int add(struct Operands temp)
{
	return (temp.o1 + temp.o1);
}

int main()
{
	struct Operands op[3];
	int n;

	op[0].o1 = 1;
	op[0].o2 = 1;
	op[1].o1 = 1;
	op[1].o2 = 1;

	op[2].o1 = add(op[0]);
	op[2].o2 = add(op[1]);
	
	n = op[2].o1 + op[2].o2;

	write(n);
	
	return 0;
}
