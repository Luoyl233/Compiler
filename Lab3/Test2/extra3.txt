{
	int o1;
	int o2;
}arr[10];

struct InterCodes
{

};

int add(struct Operands temp)
{
	return (temp.o1 + temp.o2);
}

int add2(struct InterCodes temp2)
{
	return (temp2.rlt.o1 + temp2.rlt.o2);
}

{

	code.rlt.o1 = 1;
	code.rlt.o2 = 2;
